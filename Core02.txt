
########## Base configuration
!
hostname Core02
!
enable secret cisco
!
username admin password 0 cisco
!
!
no ip domain-lookup
ip domain-name devnetlabs.test

!
ip ssh version 2
!
line con 0
 exec-timeout 3600 0
 logging synchronous
line aux 0
line vty 0 4
 exec-timeout 3600 0
 logging synchronous
 login local
!

########## ISP Core BGP Configuration
!
router bgp 64400
 bgp log-neighbor-changes
 network 20.20.20.20 mask 255.255.255.255
 neighbor 1.1.1.1 remote-as 64400
 neighbor 1.1.1.1 update-source Loopback0
 neighbor 11.11.11.11 remote-as 64411
 neighbor 11.11.11.11 ebgp-multihop 255
 neighbor 11.11.11.11 update-source Loopback0
 neighbor 22.22.22.22 remote-as 64422
 neighbor 22.22.22.22 ebgp-multihop 255
 neighbor 22.22.22.22 update-source Loopback0
 neighbor 33.33.33.33 remote-as 64433
 neighbor 33.33.33.33 ebgp-multihop 255
 neighbor 33.33.33.33 update-source Loopback0
