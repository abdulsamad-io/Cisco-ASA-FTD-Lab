
########## Base configuration
!
hostname PE02
!
enable secret cisco
!
username admin password 0 cisco
!
!
no ip domain-lookup
ip domain-name devnetlabs.test

!
ip ssh version 2
!
line con 0
 exec-timeout 3600 0
 logging synchronous
line aux 0
line vty 0 4
 exec-timeout 3600 0
 logging synchronous
 login local
!

########## Interface Addressing
!
interface GigabitEthernet0/0
 description CONNECTION_TO_MNSBP_LAN
 no ip address
 no shutdown
!
interface GigabitEthernet0/1
 description CONNECTION_TO_ISP_CORE01
 ip address 10.1.22.2 255.255.255.252
 no shutdown
!
interface GigabitEthernet0/2
 description CONNECTION_TO_ISP_CORE02
 ip address 10.2.22.2 255.255.255.252
 no shutdown
!
interface Loopback0
 description Loopback_Zero
 ip address 22.22.22.22 255.255.255.255
 no shutdown
!
interface Loopback1
 description Loopback_One
 ip address 192.168.22.22 255.255.255.255
 no shutdown
!


########## ISP Core BGP Configuration
!
router bgp 64422
 bgp log-neighbor-changes
 network 10.1.22.0 mask 255.255.255.252
 network 10.2.22.0 mask 255.255.255.252
 network 192.168.22.22 mask 255.255.255.255
 neighbor 1.1.1.1 remote-as 64400
 neighbor 1.1.1.1 update-source Loopback0
 neighbor 1.1.1.1 ebgp-multihop 255
 neighbor 2.2.2.2 remote-as 64400
 neighbor 2.2.2.2 ebgp-multihop 255
 neighbor 2.2.2.2 update-source Loopback0
!
ip route 1.1.1.1 255.255.255.255 10.1.22.1
ip route 2.2.2.2 255.255.255.255 10.2.22.1
!
